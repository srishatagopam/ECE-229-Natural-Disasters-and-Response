14def death_vs_gdp(natural_disaster_df, gdp_df): 

15 """ 

16 Plots a graph that shows the Total Death counts vs the GDP per capita value 

17 for disasters that caused over 5000 deaths. 

18  

19 **:natural_disaster_df: pd.DataFrame** 

20 The natural disaster dataframe. 

21  

22  

23 **:gdp_df: pd.DataFrame** 

24 The GDP per capita dataframe 

25  

26  

27 **Returns: None** 

28 """ 

29 assert isinstance(natural_disaster_df, pd.DataFrame) 

30 assert isinstance(gdp_df, pd.DataFrame) 

31 

32 # Join the disaster dataframe with GDP dataframe on Country and Year 

33 new_df = pd.merge(natural_disaster_df, gdp_df, how='left', left_on=['ISO','Year'], right_on = ['Code','Year']) 

34 # Filter out the major disasters 

35 new_df = new_df[new_df['Total Deaths'] > 5000] 

36 new_df = new_df[['Total Deaths','GDP per capita']].dropna() 

37 # Perform scatter plot 

38 plt.figure(figsize=(20, 12), dpi=80) 

39 new_df.plot.scatter(x = 'Total Deaths', y = 'GDP per capita',figsize=(8,5)) 

40 plt.title('Disaster Total Deaths Vs GDP per capita') 

41 plt.xlabel('Total Deaths') 

42 plt.ylabel('GDP per capita (dollars per person)') 

43 plt.show()

46def major_disaster_distribution(natural_disaster_df): 

47 """ 

48 Plots a world map that displays the density of the countries that major disasters took place. 

49  

50 **:natural_disaster_df: pd.DataFrame** 

51 The natural disaster dataframe 

52  

53 **Returns: gv.Polygons object** 

54 """ 

55 assert isinstance(natural_disaster_df, pd.DataFrame) 

56 # Do a count over the country 

57 country_count = natural_disaster_df.groupby('ISO').count()['Year'] 

58 # Import the geopandas map 

59 world = geopandas.read_file(geopandas.datasets.get_path('naturalearth_lowres')) 

60 

61 # Avoid same column name problem 

62 country_count = pd.DataFrame(country_count).rename(columns = {'Year':'Year_2'}) 

63 country_count['inde'] = country_count.index 

64 # Plot the count on the map 

65 world = world.merge(country_count, left_on = 'iso_a3', right_on = 'inde', how = 'left').fillna(0) 

66 ax = gv.Polygons(world, vdims =[('Year_2','# disasters'), ('name','Country'),]).opts( 

67 tools=['hover'], width=600,height=500, #projection=crs.Robinson()

69 return ax

71def map_distribution_cases(natural_disaster_df, voila=True): 

72 """ 

73 Plots a world map that displays the density of the countries that major disasters 

74 took place. Can also display multiple types of dependent features.  

75  

76 **:natural_disaster_df: pd.DataFrame** 

77 The natural disaster dataframe 

78  

79 **:voila: bool** 

80 Use to return plot that is visible in a notebook

82 **Returns: ipywidget.widget**

85 assert isinstance(natural_disaster_df, pd.DataFrame)

87 selector_opts = { 

88 '# Disasters':{'dep_var':'Year', "dep_var_2":'Year_2','feature_name':'# Disasters','group_op':'count'}, 

89 '# Deaths':{'dep_var':'Total Deaths', "dep_var_2":'TD_2','feature_name':'# Deaths','group_op':'sum'}, 

90 '# Injured':{'dep_var':'No Injured','dep_var_2':'num_inj','feature_name':'# Injured','group_op':'sum'},

93 'Reconstruction Costs':{'dep_var':'Reconstruction Costs (\'000 US$)','dep_var_2':'num_recon_costs','feature_name':'Reconstruction Costs','group_op':'sum'}, 

94 'Total Damanges':{'dep_var':'Total Damages (\'000 US$)','dep_var_2':'num_damages','feature_name':'Total Damanges','group_op':'sum'},

96 def major_disaster_map(start_time,end_time,vis_type=None): 

97 scale = 3/2+1/7 

98 width=int(300*scale) 

99 height=int(250*scale) 

100 

101 restricted_df = natural_disaster_df[(start_time.year<=natural_disaster_df['Year']) & 

102 (natural_disaster_df['Year']<=end_time.year)] 

103 grouped = restricted_df.groupby('ISO') 

104 country_var_dict = {} 

105 for i in selector_opts.keys(): 

106 if selector_opts[i]['group_op']=='sum': 

107 country_var_dict[selector_opts[i]['dep_var']] = grouped[selector_opts[i]['dep_var']].sum() 

108 elif selector_opts[i]['group_op']=='count': 

109 country_var_dict[selector_opts[i]['dep_var']] = grouped[selector_opts[i]['dep_var']].count() 

110 

111 # Import the geopandas map 

112 world = geopandas.read_file(geopandas.datasets.get_path('naturalearth_lowres')) 

113 

114 # Avoid same column name problem 

115 columns_names = {selector_opts[vis_type_temp]['dep_var']:selector_opts[vis_type_temp]['dep_var_2'] 

116 for vis_type_temp in selector_opts.keys()} 

117 

118 country_var = pd.DataFrame(data=country_var_dict).rename( 

119 columns = columns_names 

120 ) 

121 country_var['inde'] = country_var.index 

122 # Plot the count on the map 

123 world = world.merge(country_var, left_on = 'iso_a3', right_on = 'inde', how = 'left').fillna(0) 

124 keyDims = [(selector_opts[vis_type_temp]['dep_var_2'],selector_opts[vis_type_temp]['feature_name']) 

125 for vis_type_temp in selector_opts.keys()] 

126 keyDims = [('name','Country')]+ keyDims 

127 ax = gv.Polygons(world, vdims =keyDims).opts( 

128 tools=['hover'], 

129 width=width, 

130 height=height, 

131 color_index=selector_opts[vis_type]['dep_var_2'], 

132 colorbar=True, 

133 title="Where are disasters concentrated?"

135 bar = hv.Bars(country_var.sort_values(selector_opts[vis_type]['dep_var_2'],

137 ('inde','Country'), 

138 (selector_opts[vis_type]['dep_var_2'],selector_opts[vis_type]['feature_name'])).opts( 

139 tools=['hover'], 

140 width=width, 

141 height=height, 

142 xrotation=70, 

143 title="Which countries are affected the most?" 

144 ) 

145 

146 return ax+bar
